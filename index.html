ps_mem package repository

Latest: v1.0.3

APT (Debian / Ubuntu)

curl -fsSL https://rushikeshsakharleofficial.github.io/ps_mem-c/public.gpg | sudo gpg --dearmor -o /usr/share/keyrings/ps-mem.gpg
echo "deb [signed-by=/usr/share/keyrings/ps-mem.gpg] https://rushikeshsakharleofficial.github.io/ps_mem-c/apt /" | sudo tee /etc/apt/sources.list.d/ps-mem.list
sudo apt-get update && sudo apt-get install -y ps-mem

DNF / YUM (RHEL / Rocky / Fedora)

sudo rpm --import https://rushikeshsakharleofficial.github.io/ps_mem-c/public.gpg
sudo tee /etc/yum.repos.d/ps-mem.repo <<'REPO'
[ps-mem]
name=ps_mem
baseurl=https://rushikeshsakharleofficial.github.io/ps_mem-c/rpm/$basearch/
enabled=1
gpgcheck=1
gpgkey=https://rushikeshsakharleofficial.github.io/ps_mem-c/public.gpg
REPO
sudo dnf install -y ps_mem
